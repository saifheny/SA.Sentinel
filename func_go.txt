function go(viewId) {
  sessionStorage.setItem("currentView", viewId);
  document.querySelectorAll(".view").forEach(function (s) {
    s.classList.remove("active");
  });
  var el = document.getElementById("view-" + viewId);
  if (!el) return;
  el.classList.add("active");
  window.scrollTo(0, 0);
}
